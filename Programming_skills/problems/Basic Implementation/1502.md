impl Solution {
    pub fn can_make_arithmetic_progression(mut arr: Vec<i32>) -> bool {
        arr.sort();
        let diff:i32=arr[1]-arr[0];
        for i in 2..arr.len() {
            if arr[i]-arr[i-1]!=diff { return false; }
        }
        true
    }
}

impl Solution {
    pub fn can_make_arithmetic_progression(mut arr: Vec<i32>) -> bool {
        arr.sort_unstable();
        let mut it = arr.windows(2).map(|w| w[1]-w[0]);
        let curr = it.next().unwrap_or(0);
        it.all(|v| v==curr)
    }
