impl Solution {
    pub fn is_monotonic(nums: Vec<i32>) -> bool {
        if nums.len()==1 { return true; }
        let mut sign:i32=0;
        for i in 1..nums.len() {
            let diff=nums[i]-nums[i-1];
            if sign==0 {
                if diff>0 { sign=1; }
                else if diff<0 { sign=-1; }
            } else {
                if diff>0&&sign==-1 { return false; }
                else if diff<0&&sign==1 { return false; }
            }
        }
        true
    }
}
