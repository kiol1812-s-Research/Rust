impl Solution {
    pub fn plus_one(digits: Vec<i32>) -> Vec<i32> {
        let len:usize=digits.len();
        let mut res:Vec<i32>=Vec::new();
        let mut carry:i32=1;
        for i in 0..len {
            res.push(carry+digits[len-i-1]);
            if res[i]>=10 {
                res[i]=0;
                carry=1;
            }else{
                carry=0;
            }
        }
        if carry==1 { res.push(1); }
        res.reverse();
        res
    }
}
